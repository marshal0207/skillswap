// SECTION: Navigation
function showSection(id) {
  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
  document.getElementById(id).classList.add('active');

  if (id === 'swaps') showSwaps();
  if (id === 'home') loadHomeStats();
  if (id === 'admin') showAdminData();
}

// SECTION: Save Profile
document.getElementById('profileForm').addEventListener('submit', function (e) {
  e.preventDefault();
  const form = e.target;
  const user = {
    id: Date.now(),
    name: form.name.value.trim(),
    location: form.location.value.trim(),
    photo: form.photo?.value.trim() || '', // Add photo field in HTML
    skillsOffered: form.skillsOffered.value.split(',').map(s => s.trim()),
    skillsWanted: form.skillsWanted.value.split(',').map(s => s.trim()),
    availability: form.availability.value,
    isPublic: form.isPublic.checked,
    swaps: [],
    feedback: [],
    banned: false
  };

  localStorage.setItem('myProfile', JSON.stringify(user));
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  db = db.filter(u => u.name !== user.name); // avoid duplicates
  db.push(user);
  localStorage.setItem('userDB', JSON.stringify(db));
  alert("Profile Saved!");
  loadHomeStats();
});

// SECTION: Browse Users
document.getElementById('searchSkill').addEventListener('input', function (e) {
  const query = e.target.value.toLowerCase();
  const all = JSON.parse(localStorage.getItem('userDB')) || [];
  const me = JSON.parse(localStorage.getItem('myProfile'));
  const result = all.filter(u =>
    u.name !== me.name &&
    u.isPublic &&
    !u.banned &&
    u.skillsOffered.join().toLowerCase().includes(query)
  );
  const userList = document.getElementById('userList');
  userList.innerHTML = "";

  result.forEach(u => {
    const div = document.createElement('div');
    div.className = "user-card";
    div.innerHTML = `
      <h3>${u.name}</h3>
      <p><strong>Location:</strong> ${u.location || '-'}</p>
      ${u.photo ? `<img src="${u.photo}" alt="Profile Photo" class="profile-photo">` : ''}
      <p><strong>Offers:</strong> ${u.skillsOffered.join(', ')}</p>
      <p><strong>Wants:</strong> ${u.skillsWanted.join(', ')}</p>
      <p><strong>Availability:</strong> ${u.availability}</p>
      <button onclick="requestSwap('${u.name}')">Request Swap</button>
    `;
    userList.appendChild(div);
  });
});

// SECTION: Request Swap
function requestSwap(toUser) {
  const me = JSON.parse(localStorage.getItem('myProfile'));
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  const receiver = db.find(u => u.name === toUser);
  if (!receiver || receiver.banned) return alert("User not available.");

  const note = prompt("Add a message for your swap request (optional):");
  const swapRequest = {
    from: me.name,
    to: receiver.name,
    skillsOffered: me.skillsOffered,
    skillsWanted: me.skillsWanted,
    note: note || '',
    status: 'Pending',
    timestamp: new Date().toLocaleString()
  };

  db.forEach(u => {
    if (u.name === me.name || u.name === receiver.name) {
      u.swaps = u.swaps || [];
      u.swaps.push(swapRequest);
    }
  });

  localStorage.setItem('userDB', JSON.stringify(db));
  alert(`Swap requested to ${toUser}`);
  showSwaps();
}

// SECTION: Swap Requests
function showSwaps() {
  const me = JSON.parse(localStorage.getItem('myProfile'));
  let db = JSON.parse(localStorage.getItem('userDB')) || [];

  const incomingDiv = document.getElementById('incomingSwaps');
  const outgoingDiv = document.getElementById('outgoingSwaps');
  const statusFilter = document.getElementById('swapStatusFilter')?.value || 'all';
  const search = document.getElementById('swapSearch')?.value.toLowerCase() || '';

  incomingDiv.innerHTML = "";
  outgoingDiv.innerHTML = "";

  db.forEach(user => {
    // Incoming
    user.swaps?.forEach(req => {
      if (req.to === me.name &&
        (statusFilter === 'all' || req.status.toLowerCase() === statusFilter) &&
        (req.from.toLowerCase().includes(search) || req.skillsOffered?.join().toLowerCase().includes(search))
      ) {
        const card = document.createElement('div');
        card.className = 'swap-card';
        card.innerHTML = `
          <div>
            <p><strong>From:</strong> ${req.from}</p>
            <p><strong>Skills Offered:</strong> ${req.skillsOffered?.join(', ')}</p>
            <p><strong>Message:</strong> ${req.note || '-'}</p>
            <p><strong>Requested On:</strong> ${req.timestamp}</p>
            <span class="swap-status ${req.status.toLowerCase()}">${req.status}</span>
          </div>
          <div class="swap-actions">
            ${req.status === 'Pending' ? `
              <button class="accept" onclick="respondSwap('${req.from}', 'Accepted')">Accept</button>
              <button class="decline" onclick="respondSwap('${req.from}', 'Rejected')">Reject</button>
            ` : ''}
          </div>
        `;
        incomingDiv.appendChild(card);
      }
    });

    // Outgoing
    if (user.name === me.name) {
      user.swaps?.forEach(req => {
        if ((statusFilter === 'all' || req.status.toLowerCase() === statusFilter) &&
          (req.to.toLowerCase().includes(search) || req.skillsWanted?.join().toLowerCase().includes(search))) {
          const card = document.createElement('div');
          card.className = 'swap-card';
          card.innerHTML = `
            <div>
              <p><strong>To:</strong> ${req.to}</p>
              <p><strong>Skills Wanted:</strong> ${req.skillsWanted?.join(', ')}</p>
              <p><strong>Message:</strong> ${req.note || '-'}</p>
              <p><strong>Requested On:</strong> ${req.timestamp}</p>
              <span class="swap-status ${req.status.toLowerCase()}">${req.status}</span>
            </div>
            <div class="swap-actions">
              ${req.status === 'Pending' ? `
                <button class="decline" onclick="deleteSwap('${req.to}')">Cancel</button>
              ` : req.status === 'Accepted' && !req.feedback ? `
                <button class="accept" onclick="giveFeedback('${req.to}')">Feedback</button>
              ` : req.feedback ? `<p>✔️ Feedback Sent</p>` : ''}
            </div>
          `;
          outgoingDiv.appendChild(card);
        }
      });
    }
  });
}

function respondSwap(fromUser, action) {
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  const me = JSON.parse(localStorage.getItem('myProfile'));
  db.forEach(user => {
    user.swaps?.forEach(s => {
      if (s.to === me.name && s.from === fromUser && s.status === 'Pending') {
        s.status = action;
      }
      if (user.name === fromUser && s.to === me.name && s.status === 'Pending') {
        s.status = action;
      }
    });
  });
  localStorage.setItem('userDB', JSON.stringify(db));
  showSwaps();
}

function giveFeedback(toUser) {
  const feedback = prompt("Leave your feedback:");
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  const me = JSON.parse(localStorage.getItem('myProfile'));

  db.forEach(u => {
    if (u.name === me.name) {
      u.swaps?.forEach(req => {
        if (req.to === toUser && req.status === 'Accepted' && !req.feedback) {
          req.feedback = feedback || 'Positive';
        }
      });
    }
  });

  localStorage.setItem('userDB', JSON.stringify(db));
  alert("Thanks for your feedback!");
  showSwaps();
}

function deleteSwap(toUser) {
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  const me = JSON.parse(localStorage.getItem('myProfile'));

  db.forEach(user => {
    if (user.name === me.name) {
      user.swaps = user.swaps.filter(req => req.to !== toUser || req.status !== 'Pending');
    }
    if (user.name === toUser) {
      user.swaps = user.swaps.filter(req => req.from !== me.name || req.status !== 'Pending');
    }
  });

  localStorage.setItem('userDB', JSON.stringify(db));
  alert("Swap request cancelled");
  showSwaps();
}

// SECTION: Admin Functions
function enableAdmin() {
  const pass = document.getElementById('adminPass').value;
  if (pass === "admin123") {
    document.getElementById('adminPanel').style.display = 'block';
    showAdminData();
  } else {
    alert("Incorrect password.");
  }
}

function showAdminData() {
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  const output = document.getElementById('adminOutput');
  output.innerHTML = "<h3>All Users & Swaps</h3>";

  db.forEach(u => {
    output.innerHTML += `
      <div class="user-card">
        <strong>${u.name}</strong> ${u.banned ? '<span style="color:red;">(Banned)</span>' : ''}<br>
        Skills: ${u.skillsOffered.join(', ')}<br>
        Swaps: ${u.swaps?.length || 0}
        <button onclick="banUser('${u.name}')">Ban</button>
        <button onclick="rejectSkills('${u.name}')">Reject Skills</button>
      </div>
    `;
  });
}

function banUser(name) {
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  db.forEach(u => {
    if (u.name === name) u.banned = true;
  });
  localStorage.setItem('userDB', JSON.stringify(db));
  showAdminData();
  alert(`${name} has been banned.`);
}

function rejectSkills(name) {
  let db = JSON.parse(localStorage.getItem('userDB')) || [];
  db.forEach(u => {
    if (u.name === name) u.skillsOffered = ["(Rejected by Admin)"];
  });
  localStorage.setItem('userDB', JSON.stringify(db));
  showAdminData();
  alert(`Skills for ${name} have been rejected.`);
}

function sendAnnouncement() {
  const msg = prompt("Enter announcement message:");
  if (msg) {
    localStorage.setItem('announcement', "📢 " + msg);
    loadHomeStats();
    alert("Announcement sent!");
  }
}

function downloadData() {
  const data = localStorage.getItem('userDB') || '[]';
  const blob = new Blob([data], { type: 'application/json' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = "user_swap_data.json";
  a.click();
}

// SECTION: Home Stats & Featured Users
function loadHomeStats() {
  const db = JSON.parse(localStorage.getItem('userDB')) || [];

  document.getElementById('userCount').textContent = db.length;
  const totalSwaps = db.reduce((sum, u) => sum + (u.swaps?.length || 0), 0);
  document.getElementById('swapCount').textContent = totalSwaps;

  const announcement = localStorage.getItem('announcement') || "📢 No announcements yet";
  document.getElementById('announcementBar').textContent = announcement;

  const featured = db.filter(u => u.isPublic && !u.banned);
  const sample = featured.sort(() => 0.5 - Math.random()).slice(0, 2);
  const container = document.getElementById('featuredUsers');
  container.innerHTML = "";

  sample.forEach(user => {
    const div = document.createElement('div');
    div.className = "feature";
    div.innerHTML = `
      <h4>${user.name}</h4>
      ${user.photo ? `<img src="${user.photo}" alt="Profile Photo" class="profile-photo">` : ''}
      <p><strong>Offers:</strong> ${user.skillsOffered.join(', ')}</p>
      <p><strong>Wants:</strong> ${user.skillsWanted.join(', ')}</p>
    `;
    container.appendChild(div);
  });
}

// Swap tab toggle
function toggleSwapView(type) {
  document.querySelectorAll('.swap-tabs button').forEach(btn => btn.classList.remove('active'));
  document.querySelectorAll('.swap-list').forEach(div => div.classList.remove('active'));

  if (type === 'incoming') {
    document.querySelector('.swap-tabs button:nth-child(1)').classList.add('active');
    document.getElementById('incomingSwaps').classList.add('active');
  } else {
    document.querySelector('.swap-tabs button:nth-child(2)').classList.add('active');
    document.getElementById('outgoingSwaps').classList.add('active');
  }
}
